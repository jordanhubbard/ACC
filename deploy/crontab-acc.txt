# Emits WARN when free space < 1 GiB, CRIT when < 512 MiB.
*/15 * * * * root bash WORKSPACE/scripts/cifs-mount-health.sh --mount-path /home/jkh/.acc/shared >> /var/log/accfs-health.log 2>&1
